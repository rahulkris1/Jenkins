pipeline {
	agent any
	tools{node js "MyNode"}	
	stages {
		stage("Check Node Version"){
			steps {
			cmd "node --version"
}
}
		stage("install dependencies"){
			steps {
				cmd "npm --version"
                                cmd "npm install"
}
}
		stage("Test"){
			steps {
				cmd "node app.js"
}
}
		stage("Release the version"){
			steps {
				cmd "Release the Version"
}
}
}
}
